sub GetSpecFieldBySpecKeyFromSpecTable {
	local($mode,$tname,$key_fldname,$key_fldvalue,$fldname) = @_;
 
	if($mode eq 1) {
		&ExecSQL("SELECT $fldname FROM $tname WHERE $key_fldname = '$key_fldvalue'");
 
		#変数格納
		$ref = $sth->fetchrow_hashref();
		$strtmp = $ref->{$fldname};
		$rc = $sth->finish;
 

		return $strtmp;
	}
 

}
