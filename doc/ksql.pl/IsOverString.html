IsOverString --- バイト数を超えてないか確認 (formを参照)

渡し値(string)tname:テーブル名
戻り値(boolean)1=超えてる,0=超えてない

sub IsOverString {
	local($tname) = @_;
 
	#フィールド名をヘッドに挿入
	&ExecSQL("DESC $tname"); #ＳＱＬ文実行
	@fldname = &MakeArrayBySpecCat("Field");
		
	for($i = 0; $i <= $#fldname; $i++) {
		if(length($form{"$fldname[$i]"}) > &GetByteFromFieldType(&GetFieldType(1,$tname,$fldname[$i]))) {
			$pOverString = $fldname[$i];
			return 1;
		}
	}
 
	return 0;
}

#ＳＱＬ接続 ▼▼▼▼▼▼▼
&pConnectSQL;
 

#入力文字列のバイト数のチェック ---
if(&IsOverString($INI{'GROBAL-SiDBName'}) eq 1) {
print &Err_Kanri("$pOverString エラー");
}
