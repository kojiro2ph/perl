sub GetByteFromFieldType {
	local($fldtype) = @_;
 
	if($fldtype =~ /varchar/) {
		$fldtype =~ /varchar\((.*)\)/;
		$strtmp = $1;
	}
	elsif($fldtype =~ /text/) {
		$strtmp = 5000;
	}
 
	return $strtmp;
}
