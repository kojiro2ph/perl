sub SetFieldToArray {
	local(@strarray) = @_;
	local($TMP);
 
	while ($ref = $sth->fetchrow_hashref()) {
		foreach $TMP (@strarray) {
			push(@{$TMP},$ref->{$TMP});
		}
	}
}
