sub GetCookie {
	local($strtmp,$c,$p,$n,$v,$vn,$vv,@p,@parray);
	$c = $ENV{'HTTP_COOKIE'};
	@parray = split(/;/,$c);
	foreach $p (@parray) {
		($n,$v) = split(/=/,$p);
		$n =~ s/ //g;
		@varray = split(/,/,$v);
		foreach $v (@varray) {
			($vn,$vv) = split(/:/,$v);
			$strtmp = $n . "x" . $vn;
			${$strtmp} = $COOKIE{"$n\-$vn"} = $vv;
		}
	}
}
