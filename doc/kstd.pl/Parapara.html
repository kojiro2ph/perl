Parapara --- テーブルのリスト作成時の背景色交換関数

渡し値(string)color:色
戻り値RGB文字列 (#FFFFF or $color)

sub Parapara {
 
	local($color) = @_;
 
	if($Parapara_i eq "") {
		$Parapara_i = 1;
	}
 
	if($Parapara_i eq "1") {
		$color = "#FFFFFF";
		$Parapara_i = "0";
	} else {
		$Parapara_i = "1";		
	}
 
	return $color;
}

&ExecSQL($q);
