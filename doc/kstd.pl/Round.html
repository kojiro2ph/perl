Round --- 四捨五入する

渡し値(integer)num:数字
(integer)decimals:四捨五入する桁
戻り値(integer)四捨五入した数字

sub Round {
  local($num, $decimals) = @_;
  local($format, $magic);
  $format = '%.' . $decimals . 'f';
  $magic = ($num > 0) ? 0.5 : -0.5;
  sprintf($format, int(($num * (10 ** $decimals)) + $magic) /
                   (10 ** $decimals));
}
