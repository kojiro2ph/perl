RecordFileData --- ファイル書込処理

渡し値(string)fname:ファイル名（パス抜き）
(string)way:オプション1:各文字列配列に改行を付加して書き込む 2:各文字列配列をそのまま書き込む 3:文字列をそのまま書き込む
(string)strline:文字列変数
(string array)strArray:文字列配列
戻り値なし

sub RecordFileData {
	local($fname,$way,$strline,@strArray) = @_;
	local($strtmp);
 

	
	if($way eq 1) {
		open(DB,">$fname");
		foreach $strtmp (@strArray) {
			if($strtmp !=~ /\n$/) {
				$strtmp = join("",$strtmp,"\n");
			}
			print DB "$strtmp";	
		}
		close(DB);
	} elsif($way eq 2) {
		open(DB,">$fname");
		$strtmp = join("",@strArray);
		print DB "$strtmp";	
		close(DB);
	} elsif($way eq 3) {
		open(DB,">$fname");
		print DB "$strline";	
		close(DB);
	} elsif($way eq 4) {
		open(DB,">>$fname");
		print DB "$strline";	
		close(DB);
	}
 
	return;
}
