IchiZero --- １と０を交換する

渡し値(integer)q:1か0
戻り値(integer)q=1だったら0,q=0だったら1

sub IchiZero {
	local($q) = @_;
 
	if($q) {
		return "0";
	} else {
		return "1";
	}
}
