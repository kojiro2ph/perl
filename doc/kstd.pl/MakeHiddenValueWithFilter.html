MakeHiddenValueWithFilter --- 連想配列を<input type=hidden..に作成する

渡し値(string)remove:"v:v"フォーマットのhiddenにしないキー
(hash)hash:連想配列
戻り値(string)strtmp:作成した文字列

sub MakeHiddenValueWithFilter {
	local($remove, %hash,) = @_;
	local($strtmp, $key, $value, $tmp, $rem_flag, @removes);
 

	@removes = split(/:/, $remove);
 
	while(($key, $value) = each %hash) {
 
		$rem_flag = 0;
 
		foreach $tmp (@removes) {
			if($tmp eq $key) {
				$rem_flag = 1;
				next;
			}
		}
 
		if($rem_flag ne 1) {
			$strtmp = $strtmp . "<input type=\"hidden\" name=\"$key\" value=\"$value\">\n";
		}
	}
 

	return $strtmp;
}
