ConvPrice --- 全角の数字を半角の数字に変換する

渡し値(string)strtmp:変換する文字列
戻り値(integer)strtmp:変換した数値

sub ConvPrice {
	local($strtmp) = @_;
 

 
	$strtmp =~ s/１/1/g;
	$strtmp =~ s/２/2/g;
	$strtmp =~ s/３/3/g;
	$strtmp =~ s/４/4/g;
	$strtmp =~ s/５/5/g;
	$strtmp =~ s/６/6/g;
	$strtmp =~ s/７/7/g;
	$strtmp =~ s/８/8/g;
	$strtmp =~ s/９/9/g;
	$strtmp =~ s/０/0/g;
 
	$strtmp =~ s/\D//g;	
 
	if($strtmp eq "") {
		$strtmp = "0";
	}
 

	return $strtmp;
}
