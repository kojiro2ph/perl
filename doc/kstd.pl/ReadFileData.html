ReadFileData --- ファイル読込処理

渡し値(string)fname:ファイル名（パス抜き）
(string)way:オプション1:文字列で返す2:配列で返す3:文字列で返す
戻り値(string or string array)ファイルの文字列

sub ReadFileData {
	local($fname,$way) = @_;
	local(@strarray,$strline,$strtmp,$DebugPath);
 
	open(DB,"$fname");
	@strarray = <DB>;
	close(DB);
 
	if($way eq 1) {
		$strline = join("",@strarray);
		return "$strline";
	} elsif($way eq 2) {
		return @strarray;
	} elsif($way eq 3) {
		$strline = join("",@strarray);
		return "$strline";
	}
}
