sub InitFromString {
 
	local($strtmp) = @_;
 
	($dum,$q) = split(/\?/,$strtmp, 2);
	@qs = split(/&/, $q);
	foreach $tmp (@qs) {
		($k,$v) = split(/=/, $tmp);
		$v =~ s/%([A-Fa-f0-9][A-Fa-f0-9])/pack("C", hex($1))/eg;
		$form{$k} = $v;
	}
}
