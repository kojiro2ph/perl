SortInOne --- 配列中の重複をなくす

渡し値(string array)strArray:重複をなくす配列
戻り値(string array)tmpArray:重複をなくした配列

sub SortInOne {
	local(@strArray) = @_;
	local(@tmpArray,$flag,$i,$pcnt,$j);
 
	$flag = 0;
	# $pcnt = 0;
 
		for($i = 0; $i <= $#strArray; $i++) {
			$flag = 0;	
			if(@tmpArray eq "") {
				$tmpArray[0] = $strArray[$i];
			} else {
				for($j = 0; $j <= $#tmpArray; $j++) {
					if($strArray[$i] eq $tmpArray[$j]) {
						$flag = 1;
					break;
					}
				}
 
				if($flag eq 0) {
					$tmpArray[$#tmpArray + 1] = $strArray[$i];
				}
			}
		}
 
	return @tmpArray;
}
