SandWitch --- 文字をはさむ

渡し値(string)mode:オプション1:標準
(string)str:中央の文字
(string)substr:端の文字
戻り値(string)strtmp:文字列

sub SandWitch {
	local($mode,$str,$substr) = @_;
	local($strtmp);
 

	if($mode eq 1) {
		$strtmp = join("",$substr,$str,$substr);
	}
 
	return $strtmp;
}
