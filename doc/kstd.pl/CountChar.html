CountChar --- 指定文字カウント処理

渡し値(string)mode:オプション1:標準
(string)str:探すところの文字列
(charF)str:カウントする文字
戻り値(integer)一致数

sub CountChar {
	local($mode,$str,$charF) = @_;
	local($i,$cnt,@strarray);
 
	if($mode eq 1) {
		$cnt = 0;
		@strarray = split(//,$str);
 
		for($i = 0; $i <= $#strarray; $i++) {
			if($strarray[$i] eq $charF) {
				$cnt++;
			}
		}
 
		$strtmp = $cnt;
	}
 
	return $strtmp;
}
