GetMaxNumberFromINIHash --- 設定ファイルの指定セクションの最大値を取得する

渡し値(integer)mode:オプション
(string)section:セクション
(hash)hash:設定ファイルデータの連想配列
戻り値(integer)strtmp:最大値

sub GetMaxNumberFromINIHash {
	local($mode, $section, %hash) = @_;
	local($i,$tmp01,$strtmp);
 

	if($mode eq 1) {
		for($i = 1;$i <= 1000; $i++) {
			$tmp01 = join("",$section,"-",$i);
			if($hash{$tmp01} eq "") {
				$strtmp = $i - 1;
				last;
			}
		}
 
		return $strtmp;
	}
