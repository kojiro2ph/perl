MakeHashFromStrArray --- 配列から Hash を作成する

渡し値(string array)strarray:k=vの配列
戻り値(hash)連想配列

sub MakeHashFromStrArray {
	local(@strarray) = @_;
	local($i,$k,$v,%hash);
 

	for($i = 0; $i <= $#strarray; $i++) {
		($k,$v) = split(/=/,$strarray[$i]);
		$hash{$k} = $v;
	}
 
	return %hash;
}
