CheckIsNULLInHashWithFilter --- 連想配列の中での空チェック

渡し値(string array)remove:チェックしないキー
(hash)hash:連想配列
戻り値(integer)1 or 0

sub CheckIsNULLInHashWithFilter {
	local(*remove,*hash) = @_;
	local($k,$v);
 
	while(($k,$v) = each %hash) {
		if(&StringMatchToArray($k,@remove) ne 1) {
			if($v eq "") {
				$pNULL = $k;
				return 1;
			}
		}
	}
 
	return 0;
}
