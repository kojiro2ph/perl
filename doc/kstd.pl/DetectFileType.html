DetectFileType --- ファイルタイプ判別

渡し値(string)strtmp:ファイル名(拡張子込み)
戻り値(string)strline:拡張子情報

sub DetectFileType {
	local($strtmp) = @_;
	local($fname,$fext);
 
	($fname,$fext) = split(/./,$strtmp);
 
	if($fext eq 'gif'){
		$strline = 'image/gif';
	}
	elsif(($fext eq 'jpeg') or ($f_type eq 'jpg')){
		$strline = 'image/jpeg';
	}
	elsif($fext eq 'bmp'){
		$strline = 'image/bmp';
	}
	else{
		$strline = 'application/octet-stream';
	}
 
	return $strline;
}
