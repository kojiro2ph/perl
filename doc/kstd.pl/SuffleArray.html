SuffleArray --- 配列を混ぜる

渡し値(string array)old:混ぜる配列
戻り値(string array)new:混ぜた配列

sub SuffleArray {
	local(@old) = @_;
	local(@new);
 

	while (@old) {
		push(@new, splice(@old, int(rand() * $#old), 1));
	}
 
	return @new;
 
}
