sub ConvCSVString {
	local($strtmp) = @_;
 
	$strtmp =~ s/"/""/g;
	if($strtmp =~ /,/) {
		$strtmp = &SandWitch(1,$strtmp,"\"");
	}
 
	$strtmp =~ s/\r\n/\n/g;
	$strtmp =~ s/\n/
/g;
 
	#$strtmp =~ s/&/&/g;
	#$strtmp =~ s/"/"/g;
	#$strtmp =~ s/</</g;
	#$strtmp =~ s/>/>/g;
	#$strtmp =~ s/,/,/g;
 
	return $strtmp;
}
