sub MakeCSVHtmlTable {
	local($mode,%csv) = @_;
	local($strtmp,$i,$tmp01,$tmp02);
 
	if($mode eq 1) {
		$strtmp .= "<table border='1'>";
		for($i = 0;$i <= $csv{'MAXRECORD'}; $i++) {
			$tmp01 = join("",MAX,"_",$i);
			$strtmp .= "<tr>";	
			for($j = 0; $j <= $csv{$tmp01}; $j++) {
				$tmp02 = join("",$i,"_",$j);
				$csv{$tmp02} = &ConvHTMLTag($csv{$tmp02});
				$strtmp .= "<td>$csv{$tmp02}</td>";
			}
			$strtmp .= "</tr>";
		}
		$strtmp .= "</table>";
		return $strtmp;
	}
}
