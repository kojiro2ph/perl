OpenBinaryFileData --- バイナリでファイル読み込み

渡し値(string)fname:ファイル名 (パスを含む)
戻り値ファイルデータ

sub OpenBinaryFileData {
	local($fname) = @_;
	local($strline);
 
	open(BD,$fname);
	while(<BD>) {
		$strline = $strline . $_;
	}
	close(BD);
 
	return $strline;
}
