CheckAndMakeFile --- ファイルが存在しない場合は作成する

渡し値(string)fpath:ファイルパス(パス+ファイル名)
(string)fperm:ファイルのパーミッション
戻り値(boolean):1=作らない,0=作る

sub CheckAndMakeFile {
	local($fpath,$fperm) = @_;
 
	if(-e $fpath) {
		return "1";
	} else {
		#なければ作る
		mkdir($fpath,$fperm);
		return "0";
	}
}
