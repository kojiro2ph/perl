PrintMETA --- METAタグ文字列作成処理

渡し値(string)way:オプションeuc:charset を ＥＵＣコードにする
戻り値(string)strtmp:METAタグ文字列

sub PrintMETA {
	local($way) = @_;
	local($strtmp);
 
	if($way eq "euc") {
		$strtmp = q#<META HTTP-EQUIV="Content-Type" CONTENT="text/html; charset=x-euc-jp">#;
		return $strtmp;
	}
}
