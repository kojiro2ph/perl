GetFileArray --- ファイル配列を取得する

渡し値(integer)mode:オプション1:全部取得する2:".",".."を除いた状態で取得する3:ディレクトリのみ取得する
(string)path:取得するディレクトリのパス
戻り値なし

sub GetFileArray {
	local($mode,$path) = @_;
	local($i,$strtmp,@tmparray);
 
	if($mode eq 1) {
		opendir(DIR,"$path");
		@tmparray = readdir(DIR);
		closedir(DIR);
 
		return @tmparray;
	} elsif($mode eq 2) {
		opendir(DIR,"$path");
		@tmparray = grep(!/^(\.|\.\.)$/,readdir(DIR));
		closedir(DIR);
 
		return @tmparray;
	} elsif($mode eq 3) {
		opendir(DIR,"$path");
		@tmparray = grep { (-d "$path$_") && (!/^(\.|\.\.)$/) } readdir(DIR);
		closedir(DIR);
 
		return @tmparray;
	}
	
}
