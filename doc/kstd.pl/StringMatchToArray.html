StringMatchToArray --- 配列中に指定した文字列が要素内になるかどうか

渡し値(string)strF:探す文字列
(string array)strarray:配列
戻り値(boolean): 1=ある,0=ないか

sub StringMatchToArray {
 
	local($strF,@strarray) = @_;
	local($TMP);
 

	foreach $TMP (@strarray) {
		if(($TMP eq $strF) && ($strF ne "")) {
			return "1";
		}
	}
 
	return "0";
}
